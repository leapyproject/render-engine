## Install

```bash
npm install --save render-engine
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'render-engine'
import 'render-engine/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
